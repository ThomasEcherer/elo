sol.define("holme.common.ix.actions.FileNameImport", {
  extend: "sol.common.ix.ActionBase",

  cfg: undefined,

  initialize: function (config) {
    var me = this;
    me.$super("sol.common.ix.ActionBase", "initialize", [config]);
    me.cfg = config;
  },

  /**
   * Returns the name of this action.
   * @return {String}
   */
  getName: function () {
    return "FileNameImport";
  },

  /**
   * {{description}}
   */
  process: function () {
    var me = this,
        importer;
    
    importer = sol.create("holme.common.FileNameImporter", me.cfg);
    importer.process();
  }

  

});


/**
 * @member holme.common.ix.actions.FileNameImport
 * @method RF_holme_common_action_FileNameImport
 * @static
 * @inheritdoc sol.common.ix.ActionBase.RF_FunctionName
 */
function RF_holme_common_action_FileNameImport(ec, configAny) {
  var config, action, result;

  logger.enter("RF_holme_common_action_FileNameImport", configAny);

  config = sol.common.ix.RfUtils.parseAndCheckParams(ec, arguments.callee.name, configAny), 
  action = sol.create("holme.common.ix.actions.FileNameImport", config);
  result = action.execute();

  logger.exit("RF_holme_common_action_FileNameImport");

  return result;
}
